Quick Start

go mod init myapp
go get github.com/primadi/lokstra

// main.go
package main

import "github.com/primadi/lokstra"

func main() {
    ctx := lokstra.NewGlobalContext()
    app := lokstra.NewApp(ctx, "myapp", ":8080")

    app.GET("/hello", func(c *lokstra.Context) error {
        return ctx.Ok("Hello from Lokstra!")
    })

    app.Start()
}
